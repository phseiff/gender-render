16 """Bundles various methods for parsing gender*render pronoun data in a pipeline together."""

18 @staticmethod 

19 def pd_string_to_dict(pd: str) -> dict: 

20 """Parses a JSON string into a dict and returns it.""" 

21 try: 

22 pd_dict = json.loads(pd) 

23 except json.decoder.JSONDecodeError: 

24 raise errors.InvalidPDError("The given pronoun data (given as a string) is not a valid piece of JSON data.") 

25 return pd_dict

27 @staticmethod 

28 def type_of_pd(pd: Union[GRPD, IDPD]) -> Union[GRPD, IDPD]: 

29 """Returns GRPD or IDPD, depending on what type (if any) of pronoun data the given string is. 

30 Raises an error if it is neither of these two.""" 

31 if type(pd) is dict: 

32 all_values_are_strings = True 

33 all_values_are_dicts = True 

34 for value in pd.values(): 

35 if type(value) is not str: 

36 all_values_are_strings = False 

37 if type(value) is not dict: 

38 all_values_are_dicts = False 

39 if all_values_are_dicts: 

40 for idpd in pd.values(): 

41 for v in idpd.values(): 

42 if type(v) is not str: 

43 raise errors.InvalidPDError( 

44 "A property of a piece of individual pronoun data is assigned a non-string value.") 

45 if "" in pd: 

46 raise errors.InvalidPDError( 

47 "\"\" is not a valid id value in a piece of gender*render pronoun data") 

48 return GRPD 

49 elif all_values_are_strings: 

50 return IDPD 

51 else: 

52 raise errors.InvalidPDError( 

53 "The values of all top-level key-value-pairs of the given pronoun data are neither all objects, nor" 

54 + " are they all strings, but rather, a mixture of multiple types. This is neither allowed for a" 

55 + " piece of individual pronoun data (has only string values), nor a piece of gender*render pronoun" 

56 + " data (which has only object values), and is therefore invalid." 

57 ) 

58 else: 

59 raise errors.InvalidPDError( 

60 "This piece of pronoun data is neither individual pronoun data nor gender*render pronoun data." 

61 )

63 @staticmethod 

64 def return_pd_if_it_is_valid(pd: dict) -> Union[GRPD, IDPD]: 

65 """Takes a dict, raises an error if it is neither a piece of gender*render pronoun data nor a piece of 

66 individual pronoun data and returns it otherwise."""

73 @staticmethod 

74 def pd_dict_to_grpd_dict(pd: Union[GRPD, IDPD]) -> GRPD: 

75 """Converts a piece of valid pronoun data to a grpd, in case it is an idpd. 

76 The returned object may be identical to the given object or reference it.""" 

77 if GRPDParser.type_of_pd(pd) is GRPD: 

78 return pd 

79 else: # GRPDParser.type_of_pd(pd) is IDPD: 

80 return {"": pd}

82 @staticmethod 

83 def grpd_dict_to_canonical_grpd_dict(pd: GRPD) -> GRPD: 

84 """Takes a gender*render pronoun data dict and returns it with every property made canonical. 

85 Raises a DoubledInformationError if a piece of idpd contains two properties for the same attribute. 

86 Raises a InvalidInformationError if a property has a value it may not have. 

87 Raises a warning if a custom attribute is not recognizable as such.""" 

88 result = dict() 

89 for id in pd: 

90 new_idpd = dict() 

91 for gr_property, value in pd[id].items(): 

92 canonical_context_value = ContextValues.get_canonical(gr_property, is_from_tag=False) 

93 

94 # raise an error if two properties for the same attribute exist: 

95 if canonical_context_value in new_idpd: 

96 raise errors.DoubledInformationError("The individual pronoun data for id \"" + id + "\" defines " 

97 + "multiple values for the \"" + canonical_context_value 

98 + "\" attribute, using different properties. Only one is " 

99 + "allowed!") 

100 

101 # raises an error if an attribute with a limited set of valid values is used and the value is invalid: 

102 if not ContextValues.value_is_allowed(canonical_context_value, value): 

103 raise errors.InvalidInformationError("The individual pronoun data for id \"" + id + "\" defines " 

104 + "\"" + value + "\" as the value for \"" 

105 + canonical_context_value + "\" even though this attribute " 

106 + "does not allow this value.") 

107 

108 # raises a warning if a custom attribute does not use the special syntax for custom properties: 

109 if (ContextValues.is_a_custom_value(gr_property) 

110 and not ContextValues.uses_special_custom_value_syntax(gr_property)): 

111 warnings.WarningManager.raise_warning("The individual pronoun data for individual \"" + id + "\" " 

112 + "contains a custom property, but said property does not use" 

113 + " special custom property syntax.", 

114 warnings.UnknownPropertyWarning) 

115 new_idpd[canonical_context_value] = value 

116 result[id] = new_idpd 

117 return result

119 @staticmethod 

120 def full_parsing_pipeline(pd: dict) -> GRPD: 

121 """Parses a dict into a valid piece of grpd following the pipeline defined by GRPDParser, and raises an error if 

122 this turns out to be impossible.""" 

123 pd = GRPDParser.return_pd_if_it_is_valid(pd) 

124 pd = GRPDParser.pd_dict_to_grpd_dict(pd) 

125 pd = GRPDParser.grpd_dict_to_canonical_grpd_dict(pd) 

126 return pd
