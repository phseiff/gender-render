1"""Defines the warnings specified by the spec. 

2 

3To enable and disable warnings, pass a set of all warnings you want enabled to the "enable_warnings"-parameter of any 

4of the interfaces defined in the specifications. All warnings are declared as classes in this submodule. 

5 

6By default, all warnings are enabled. 

7 

8Ready-made sets of values for the "enable_warnings"-parameter include: 

9 

10- ENABLE_ALL_WARNINGS: a set of all warnings 

11- ENABLE_DEFAULT_WARNINGS: a set of all warnings enabled by default 

12- DISABLE_ALL_WARNINGS: an empty set (disables all warnings) 

13- ENABLE_ALL_LOGGING: a set of all warning classes that enable types of logging rather than warnings. 

14 

15Please note that classes that enable logging rather than actual warnings are not included in the 

16gender_render.ENABLE_ALL_WARNINGS set. Every type of warning that would be raised regardless of input at initialization 

17time of the module is considered logging rather than warning. 

18 

19When calling a function, method or class that isn't specified as a public interface by the specification (currently only 

20render_template, Template and PronounData), the warning behavior is unreliable; you therefore shouldn't do this unless 

21you explicitly define your preferred warning settings with `WarningManager.set_warning_settings`. 

22"""

151def get_all_subclasses(class_object): 

152 """Returns a list of all subclasses of a given class that are defined in the global scope.""" 

153 return [globals()[w] for w in globals() if inspect.isclass(globals()[w]) and issubclass(globals()[w], class_object)]

170# WarningManager: 

171 

172 

173class WarningManager: 

174 """A bundle of functions to handle warning handling.""" 

175 warning_settings_by_thread_id = dict() 

176 

177 @staticmethod 

178 def set_warning_settings(warning_settings: WarningSettingType): 

179 """Sets the warning settings to warning_settings for the current thread (thread-save).""" 

180 WarningManager.warning_settings_by_thread_id[threading.get_ident()] = warning_settings 

181 

182 @staticmethod 

183 def raise_warning(text: typing.Union[str, None], warning_type: WarningType): 

184 """Raises the given warning type with the given text if it is enabled for the current thread.""" 

185 if text is None: 

186 text = warning_type.__doc__ 

187 if threading.get_ident() not in WarningManager.warning_settings_by_thread_id: 

188 WarningManager.warning_settings_by_thread_id[threading.get_ident()] = ENABLE_DEFAULT_WARNINGS 

189 if warning_type in WarningManager.warning_settings_by_thread_id[threading.get_ident()]: 

190 ws.warn(text, warning_type)
