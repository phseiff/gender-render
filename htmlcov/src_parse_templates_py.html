25# definitions of words and word groups accepted by the finite state machine:

27 

28class Chars: 

29 """Helper to categorize characters.""" 

30 escape_char = "\\" 

31 whitespace_chars = "\t\n \u200B" 

32 special_chars = ":*{}\\" 

33 ws = "whitespace" 

34 char = "non-special chars" 

35 

36 @staticmethod 

37 def type(c: str) -> str: 

38 """Returns the type of character c, which determines how states in the finite state machine that describes 

39 gender*render syntax transition to each other. 

40 This is either Chars.ws (whitespace), a special character or Chars.char (anything else).""" 

41 if c in Chars.special_chars: 

42 return c 

43 elif c in Chars.whitespace_chars: 

44 return Chars.ws 

45 else: 

46 return Chars.char 

47 

48 @staticmethod 

49 def escape_gr_string(s: str, strict: bool = True) -> str: 

50 """Escapes all special gender*render characters in a string, such as {, }, \\, : and *, as well as whitespace, 

51 with backslashs. 

52 if `strict` is set to False, only {, } and \\ are escaped; this may be used for strings that are supposed to go 

53 into gender*render templates, yet not into the inners of the tags themselves.""" 

54 i = len(s) - 1 

55 while i > -1: 

56 if s[i] in ((Chars.special_chars + Chars.whitespace_chars) if strict else {"\\", "{", "}"}): 

57 s = s[:i] + "\\" + s[i:] 

58 i -= 1 

59 return s

64class States: 

65 """Combines values for all sections the finite state machine that describes the syntax can be in, as well as 

66 methods to handle the special escaped/unescaped versions of all states.""" 

67 

68 # Currently, the read character is... 

69 not_within_tags = "...not part of any tag" 

70 

71 in_empty_section = "...in a yet empty section" 

72 in_not_empty_section = "...in a not anymore empty section" 

73 in_section_with_one_finished_word = "...in a section which already contains a finished word" 

74 

75 in_empty_value_section = "...in a yet empty value section" 

76 in_not_empty_value_section = "...in a not empty value section" 

77 

78 escaped = "...and escaped"

80 @staticmethod 

81 def escape(state: str) -> str: 

82 """Converts an unescaped state to its escaped equivalent.""" 

83 assert not States.is_escaped(state) 

84 return state + States.escaped 

85 

86 @staticmethod 

87 def unescape(state: str) -> str: 

88 """Convert an escaped state to its unescaped equivalent.""" 

89 assert States.is_escaped(state) 

90 return state[:len(state)-len(States.escaped)] 

91 

92 @staticmethod 

93 def is_escaped(state: str) -> bool: 

94 """Checks is the current char is an escaped char.""" 

95 return state.endswith(States.escaped) 

96 

97 @staticmethod 

98 def switch_escapement(state: str) -> str: 

99 """Returns the escaped or unescaped state of the given state, depending on whether it is currently escaped or 

100 not.""" 

101 return States.escape(state) if not States.is_escaped(state) else States.unescape(state)

176 @staticmethod 

177 def transition_state(state: str, char: str) -> Tuple[str, Callable[[ParsedTemplate, str], ParsedTemplate]]: 

178 """For a given state s and a given character c, returns the next state s2 and a function that takes a 

179 list representation of the already-parsed data and c and returns a modified, extended duplicate of the data 

180 based on c.""" 

181 type_of_char = Chars.type(char) 

182 if type_of_char in StateTransitioner.state_transitions[state]: 

183 return StateTransitioner.state_transitions[state][type_of_char] 

184 else: 

185 raise errors.SyntaxError("Parsing error: \"" + type_of_char + "\" may not occur if it is " + state[3:])

199 @staticmethod 

200 def section_type_accepts_multiple_values(section_type: str) -> bool: 

201 """Checks whether a section type can have multiple whitespace-separated values.""" 

202 return SectionTypes.section_type_exists(section_type) and bool( 

203 [i for i in range(len(SectionTypes.section_types_w_priorities)) 

204 if SectionTypes.section_types_w_priorities[i][0] == section_type 

205 and SectionTypes.section_types_w_priorities[i][2] is True] 

206 ) 

207 

208 @staticmethod 

209 def section_type_exists(section_type: str) -> bool: 

210 """Checks if a section type exists.""" 

211 return bool( 

212 [i for i in range(len(SectionTypes.section_types_w_priorities)) 

213 if SectionTypes.section_types_w_priorities[i][0] == section_type] 

214 )

216 @staticmethod 

217 def create_section_types_for_untyped_tag(section_types: List[str]) -> List[str]: 

218 """Receives a list of section types in a tag (in chronological order) and assigns section types to those 

219 section without a section type, in accordance with the priorities of section types and the specification. 

220 Returns the typed section type list. 

221 Raises errors if section matching can not be done, or if no context section could be found.""" 

222 result = list() 

223 

224 # get all explicitly specified section types into a set: 

225 already_used = set() 

226 if len(section_types) > len(SectionTypes.section_types_w_priorities): 

227 raise errors.SyntaxPostprocessingError("Tag contains more sections than there are section types.") 

228 for section_type in filter(lambda x: x != "", section_types): 

229 if section_type in already_used: 

230 raise errors.SyntaxPostprocessingError("Section type \"" + section_type + "\" used twice in a tag.") 

231 elif not SectionTypes.section_type_exists(section_type): 

232 raise errors.SyntaxPostprocessingError("Section type \"" + section_type + "\" does not exist.") 

233 else: 

234 already_used.add(section_type) 

235 

236 # create a section priority queue without these element: 

237 available_sections_types = [s for s in SectionTypes.section_types_w_priorities if s[0] not in already_used] 

238 available_sections_types.sort(key=lambda s: s[1]) 

239 

240 # iterate over all declared section types from the left to the right: 

241 for section_type in reversed(section_types): 

242 if section_type == "": 

243 result.insert(0, available_sections_types.pop()[0]) 

244 else: 

245 result.insert(0, section_type) 

246 

247 # raise an error if there is no context value: 

248 if "context" not in result: 

249 raise errors.SyntaxPostprocessingError("Tag misses a \"context\"-section.") 

250 

251 return result

257class GRParser: 

258 """Unites several static methods of a pipeline for parsing gender*render templates from strings into a list format 

259 and refining this representation to the maximum extend possible without additionally seeing the corresponding 

260 gender*render pronoun data. 

261 

262 These functions are written to be executed in the order they are called by `full_parsing_pipeline`, and may or may 

263 not behave as expected when called on a value that didn't go through the other functions first.""" 

264 

265 @staticmethod 

266 def parse_gr_template_from_str(template: str) -> ParsedTemplate: 

267 """Takes a gender*render template as a string and returns it as an easily readable list representation. 

268 This does only do syntactic parsing in accordance to the defining finite state machine; 

269 further steps in the parsing pipeline are implemented by other methods of this parser. 

270 

271 The resulting output is of the following structure: 

272 * value of a section: represented by lists of strings 

273 * type of section: represented by a string 

274 * section: tuple of type representation and value representation 

275 * tag: list of section representation 

276 * template: list, where every uneven element represents a tag and every even element is a string 

277 

278 Special characters are all unescaped in the parsed version of the template.""" 

279 

280 result = [""] 

281 s = States.not_within_tags 

282 line_no = 1 

283 char_no = 1 

284 # iterate over all characters: 

285 for i in range(len(template)): 

286 c = template[i] 

287 # increment char count for SyntaxError raising: 

288 if c == "\n": 

289 line_no += 1 

290 char_no = 1 

291 else: 

292 char_no += 1 

293 

294 # log: 

295 warnings.WarningManager.raise_warning( 

296 "result: " + str(result) + "\n\n" 

297 + "c: \"" + c + "\"\n" 

298 + "s: " + s + "\n" 

299 + "char type: " + Chars.type(c), 

300 warnings.GRSyntaxParsingLogging) 

301 

302 # do the work of the finite state machine: 

303 type_of_char = Chars.type(c) 

304 if States.is_escaped(s): 

305 s = States.unescape(s) 

306 s, processing_function = StateTransitioner.state_transitions[s][Chars.char] 

307 result = processing_function(result, c) 

308 else: 

309 if type_of_char == Chars.escape_char: 

310 s = States.escape(s) 

311 else: 

312 try: 

313 s, processing_function = StateTransitioner.transition_state(s, c) 

314 result = processing_function(result, c) 

315 except errors.SyntaxError: 

316 raise errors.SyntaxError( 

317 "The given gender*render template has invalid syntax.", 

318 ("unknown file", line_no, char_no, template.split("\n")[line_no - 1]) 

319 ) 

320 

321 # raise an error if the template ends unproperly: 

322 if States.is_escaped(s): 

323 raise errors.SyntaxError("The template ends with an unescaped escape character, please escape it.", 

324 ("unknown file", line_no, char_no, template.split("\n")[-1])) 

325 elif s != States.not_within_tags: 

326 raise errors.SyntaxError("A tag opens, but is not finished properly.", 

327 ("unknown file", line_no, char_no, template.split("\n")[-1])) 

328 

329 return result

331 @staticmethod 

332 def assign_types_to_all_sections(parsed_template: ParsedTemplate) -> ParsedTemplate: 

333 """Takes a parsed template (as it is created by all methods of GRParser) and assigns every section of undefined 

334 type a section type.""" 

335 result = copy.deepcopy(parsed_template) 

336 for i in range(1, len(result), 2): 

337 old_section_types: List[str] = [section[0] for section in result[i]] 

338 new_section_types: List[str] = SectionTypes.create_section_types_for_untyped_tag(old_section_types) 

339 result[i] = [(new_section_types[s], result[i][s][1]) for s in range(len(new_section_types))] 

340 return result

342 @staticmethod 

343 def split_tags_with_multiple_context_values(parsed_template: ParsedTemplate) -> ParsedTemplate: 

344 """Takes a parsed template (as it is created by all methods of GRParser) and splits every tag into a sequence of 

345 tags, one for every context value of the tag. 

346 This assumes that every section was already assigned a type by GRParser.assign_types_to_all_sections, and may 

347 lead to wrong results otherwise. 

348 The context section is left the end of the tag by this procedure.""" 

349 result = copy.deepcopy(parsed_template) 

350 for i in reversed(range(1, len(result), 2)): 

351 tag_without_context_section = [section for section in result[i] if section[0] != "context"] 

352 tag_but_only_context_section = [section for section in result[i] if section[0] == "context"] 

353 

354 # split tag into one tag for every context value: 

355 context_values = tag_but_only_context_section.pop()[1] 

356 sequence_of_tags = [ 

357 (copy.deepcopy(tag_without_context_section) + [("context", [context_value])]) 

358 for context_value in context_values 

359 ] 

360 for j in reversed(range(1, len(sequence_of_tags))): 

361 sequence_of_tags.insert(j, " ") 

362 result[i:i+1] = sequence_of_tags 

363 

364 return result

366 @staticmethod 

367 def make_sure_that_sections_dont_exceed_allowed_amount_of_values(parsed_template: ParsedTemplate) -> ParsedTemplate: 

368 """Takes a parsed template (as it is created by all methods of GRParser) and raises an error if any tag that 

369 does not allow multiple values has multiple values. This should always be used before calling 

370 convert_tags_to_indxable_dicts. 

371 Returns the given dict afterwards.""" 

372 for i in range(len(parsed_template)): 

373 if i % 2: # is a tag 

374 for section_type, section_values in parsed_template[i]: 

375 if SectionTypes.section_type_accepts_multiple_values(section_type): 

376 continue 

377 elif len(section_values) > 1: 

378 raise errors.SyntaxPostprocessingError("Tag no. " + str((i + 1) / 2) + " (\"" 

379 + ReGRParser.unparse_gr_tag(parsed_template[i]) 

380 + "\") has multiple values in \"" 

381 + section_type + 

382 "\"-section even though this type of section does" 

383 + " not support this.") 

384 return parsed_template

386 @staticmethod 

387 def convert_tags_to_indexable_dicts(parsed_template: ParsedTemplate) -> ParsedTemplateRefined: 

388 """Takes a parsed template (as it is created by all methods of GRParser) and converts every tag from a 

389 representation a la "[(a, b), (c, d)]" to a representation a la "{a: b, c: d}". 

390 This makes the value of specific types of sections easier to access by other methods. 

391 Note that the result returned by this method is different in that it is not accepted by the other methods of 

392 GRParser, and that this method should thus be the last method in this pipeline to be used. 

393 Raises an error if a section has multiple values yet accepts only one.""" 

394 result = copy.deepcopy(parsed_template) 

395 for i in range(len(result)): 

396 if i % 2: # is a tag 

397 new_tag = dict() 

398 for section_type, section_values in result[i]: 

399 if not SectionTypes.section_type_accepts_multiple_values(section_type) or section_type == "context": 

400 new_tag[section_type] = section_values[0] 

401 else: 

402 new_tag[section_type] = section_values 

403 result[i] = new_tag 

404 

405 return result

407 @staticmethod 

408 def convert_context_values_to_canonicals(parsed_template: ParsedTemplateRefined) -> ParsedTemplateRefined: 

409 """Converts a parsed template as returned by GRParser.convert_tags_to_indexable_dicts to a parsed template 

410 where every context value is canonical.""" 

411 result = copy.deepcopy(parsed_template) 

412 for i in range(1, len(parsed_template), 2): 

413 result[i]["context"] = handle_context_values.ContextValues.get_canonical(result[i]["context"]) 

414 return result

416 @staticmethod 

417 def full_parsing_pipeline(template: str) -> ParsedTemplateRefined: 

418 """Walks template through the full parsing pipeline defined by GRParser, and returns the result.""" 

419 template = GRParser.parse_gr_template_from_str(template) 

420 template = GRParser.assign_types_to_all_sections(template) 

421 template = GRParser.split_tags_with_multiple_context_values(template) 

422 template = GRParser.make_sure_that_sections_dont_exceed_allowed_amount_of_values(template) 

423 template = GRParser.convert_tags_to_indexable_dicts(template) 

424 template = GRParser.convert_context_values_to_canonicals(template) 

425 return template

444class ReGRParser: 

445 """Bundles methods to get a valid gender*render template from ParsedTemplate.""" 

446 

447 @staticmethod 

448 def unparse_gr_tag(tag_representation: List[Tuple[str, List[str]]]) -> str: 

449 return "{" + "*".join([( 

450 ((Chars.escape_gr_string(section[0]) + ":") if section[0] else "") 

451 + " ".join([Chars.escape_gr_string(value) for value in section[1]]) 

452 ) for section in tag_representation]) + "}" 

453 

454 @staticmethod 

455 def unparse_gr_template(parsed_template: ParsedTemplate) -> str: 

456 """Takes the result of any method of the GRParser class and returns a template (as a string) that corresponds to 

457 the given parsed template. 

458 This may be used for testing purposes or to simplify gender*render templates.""" 

459 result = str() 

460 for i in range(len(parsed_template)): 

461 if i % 2: # is a tag 

462 result += ReGRParser.unparse_gr_tag(parsed_template[i]) 

463 else: # is a string 

464 result += Chars.escape_gr_string(parsed_template[i], strict=False) 

465 return result
