134 """Bundles several static methods to handle and create data that describes the differently gendered versions of

136 These methods form a pipeline used for creating a full linked graph of gendered nouns.

141 @staticmethod 

142 def load_from_web() -> GeneratedGenderNounData: 

143 """Creates a JSON object describing the differently gendered versions of every gendered noun. 

144 Word with no wordnet_senseno-attribute (that are not from wordnet) are ignored, and words with "other" as gender 

145 are re-gendered as "neutral". 

146 

147 The data used for this is taken from https://github.com/ecmonsen/gendered_words 

148 (which is not by me; see the repository for the license).""" 

149 

150 grammatical_genders = {"m": "male", "f": "female", "n": "neutral"} 

151 

152 # load from the web, in a completely wrong format that we have yet to change: 

153 raw_json: OriginalGenderNounData = json.loads(requests.get( 

154 "https://raw.githubusercontent.com/phseiff/gendered_words/master/gendered_words.json").text) 

155 

156 # change the format: 

157 result = dict() 

158 for word in raw_json: 

159 if "wordnet_senseno" in word: 

160 result[word["word"]] = {"gender_map": dict()} 

161 if "gender_map" in word: 

162 for short, long in grammatical_genders.items(): 

163 if short in word["gender_map"]: 

164 result[word["word"]]["gender_map"][long] = word["gender_map"][short][0]["word"].replace( 

165 " ", "_") 

166 if word["gender"] == "o": 

167 lwarn("Found an \"other\"-word! It's \"" + word["word"] + "\".") 

168 result[word["word"]]["gender"] = "neutral" 

169 else: 

170 result[word["word"]]["gender"] = grammatical_genders[word["gender"]] 

171 else: 

172 lwarn("\"" + word["word"] + "\" ignored because it is not part of wordnet and therefore not a hyponym " 

173 + "for a person.") 

174 

175 lwarn(len(result), "words found.") 

176 return result

178 @staticmethod 

179 def load_from_disk(file_name: str) -> GeneratedGenderNounData: 

180 """Loads the gendered nouns from a piece of json data contained in the given file. 

181 Returns it as a dictionary, with the only change being that the `warnings`-attribute of every noun is converted 

182 from a list to a set. 

183 This assumes the data in the given file to be of the same type as the data used throughout all the other methods 

184 of this class, NOT the format of the repository from which `load_from_web` loads its data.""" 

185 with open(file_name, "r") as f: 

186 code = f.read() 

187 code_as_a_dict = json.loads(code) 

188 for word_data in code_as_a_dict.values(): 

189 if "warning" in word_data: 

190 word_data["warning"] = set(word_data["warning"]) 

191 return code_as_a_dict

193 @staticmethod 

194 def save_to_disk(graph: GeneratedGenderNounData, file_name: str) -> None: 

195 """Saves the given gendered nouns data to the given file as json. The data is saved without any modifications, 

196 except for the `warnings`-attribute of all noun data, which is converted from a set to a list to be compatible 

197 with standard json format. 

198 This assumes the given data to be of the same type as the data used throughout all the other methods of this 

199 class, NOT the format of the repository from which `load_from_web` loads its data. 

200 The data saved with this method can be read again with `load_from_disk`, so that `save_to_disk(data, file_name)` 

201 implies `load_from_disk(file_name) == data`. 

202 This method is guaranteed to not change the given data in-place.""" 

203 graph_copy: dict = copy.deepcopy(graph) 

204 for word_data in graph_copy.values(): 

205 if "warning" in word_data: 

206 word_data["warning"] = list(word_data["warning"]) 

207 word_data["warning"].sort() 

208 with open(file_name, "w") as f: 

209 json.dump(graph_copy, f, indent=4, sort_keys=True)

212 def remove_words_that_are_not_nouns(graph: GeneratedGenderNounData) -> GeneratedGenderNounData: 

213 """Removes all elements that are not nouns from the graph. Words that are not nouns, but whose gendered versions 

214 contain nouns for some reason, are not purged by this.

216 This step should usually do nothing, since all words that are not nouns should already be filtered out when

219 

220 This step may or may not change the given object in-place."""

222 grammatical_genders = ["male", "female", "neutral"]

224 count = 0

258 @staticmethod 

259 def choose_better_word(option1, option2, log=False): 

260 """There are cases where a word has two different gendered versions for the same gender; 

261 for example, "foremother" and "ancestress" are both female words for an ancestor. In these cases, the "better" 

262 one will be chosen according to this pipeline. 

263 This is relevant for `make_all_links_two_sided`, for example (see docstring). 

264 

265 If `log` is set, having to resort to alphabetical decisions is logged.""" 

266 

267 # ToDo: Add further rules to this, or better yet, make manual (no automated data!) PRs to 

268 # https://github.com/phseiff/gendered-words to clear up all these unnecessary unclearnesses. 

269 bad_things_in_order_of_badness: List[Callable[[str], int]] = [ 

270 # the sooner the lambda, the worse it is: 

271 lambda word: len([c for c in word if c == "_"]), 

272 lambda word: int(word.endswith("man") or word.endswith("woman")), 

273 lambda word: int(word.endswith("person")), 

274 lambda word: int(word.endswith("mother") or word.endswith("father")), 

275 lambda word: int(word.endswith("parent")) 

276 ] 

277 for badness_meter in bad_things_in_order_of_badness: 

278 if badness_meter(option1) > badness_meter(option2): 

279 return option2 

280 elif badness_meter(option2) > badness_meter(option1): 

281 return option1 

282 # otherwise, return the alphabetically first one: 

283 if log: 

284 lwarn("Had to alphabetically decide between", option1, "and", option2) 

285 return sorted([option1, option2])[0]

287 @staticmethod 

288 def make_all_links_two_sided(graph: GeneratedGenderNounData, log_clashs=False)\ 

289 -> GeneratedGenderNounData: 

290 """Returns a version of the graph where every word linked to links back to the word linking to it, if this words 

291 gender is not yet in its gender mapping. 

292 This also goes for triangles where a links to b and c, but b and c are not linked. 

293 

294 If `log_clashs` is set to True, clashes are logged (these are cases where there are multiple versions available 

295 for the gendered version of a word). This should only be enabled before 

296 `create_gendered_versions_for_words_that_end_with_gender_indicators` is called on the data for the first time, 

297 since afterwards, multiple words might be connected that wheren't conntected before. One example would be that 

298 "wonder_woman" and "wonder_girl" are both considered female, and both will get "wonder_person" assigned as their 

299 neutral version by `create_gendered_versions_for_words_that_end_with_gender_indicators`. This is not an issue 

300 since one of them already links to "wonder_boy", and the other one to "wonder_man" from the beginning, but 

301 they will still be considered clashing by the algorithm, so calling `make_all_links_two_sided` with 

302 `log_clashs` set to True after applying `create_gendered_versions_for_words_that_end_with_gender_indicators` to 

303 the data will create logs for two words that don't clash more than linguistically necessary. 

304 

305 If `infect_warnings` isn't explicitely set to False, all warnings added to a word are "passed" to every word 

306 it's linked to.""" 

307 

308 count = 0 

309 

310 # makes sure we do not report a clash twice. 

311 already_reported_clashes: Set[str] = set() 

312 

313 def assign_value_or_use_old_one(gender_dict, gender, value): 

314 """Assigns a gendered version (`value`) to the given `gender` in the `gender_dict`. 

315 If there is already a value for the given gender, a decision is made whether to replace the old value in the 

316 gender dict with the newfound alternative one, and we then log the decision if `log_clashs` is set, and 

317 collect information required to build a warning to attach to the word to indicate that there was an 

318 algorithmic choose between two options.""" 

319 if gender in gender_dict and gender_dict[gender] != value: 

320 # log the clash if it wasn't logged yet: 

321 clash_report = value + " clashes with " + gender_dict[gender] 

322 if log_clashs and clash_report not in already_reported_clashes: 

323 lwarn(clash_report) 

324 already_reported_clashes.add(clash_report) 

325 

326 # choose the better of both options (keep the old word or rather use the new one): 

327 chosen_option = GenderNounDataHandler.choose_better_word(value, gender_dict[gender], log=log_clashs) 

328 else: 

329 chosen_option = value 

330 

331 gender_dict[gender] = chosen_option 

332 gender_dict_alts[gender] |= {chosen_option, value}

334 def get_all_connected_words(word_name, data: Union[Set[str], None] = None): 

335 """Returns a set of all words that are indirectly or directly linked to the given word.""" 

336 if not data: 

337 data = {word_name} 

338 for link_name in graph[word_name]["gender_map"].values(): 

339 if link_name not in data: 

340 data.add(link_name) 

341 get_all_connected_words(link_name, data) 

342 if link_name in link_groups: 

343 data |= link_groups[link_name] 

344 return data 

345 

346 link_groups: Dict[str, Set[str]] = dict() 

347 words_we_already_visited: Set[str] = set() 

348 # first iteration is to make FULL links groups of connected words before we start linking words to each other: 

349 for word_name, word_data in graph.items(): 

350 if word_name not in words_we_already_visited: 

351 

352 # create group of all words that are linked to this word (including those that are once removed): 

353 link_group = get_all_connected_words(word_name) 

354 for link_name in link_group: 

355 if link_name in link_groups: 

356 link_groups[link_name] |= link_group 

357 else: 

358 link_groups[link_name] = link_group 

359 

360 # add these words to the visited words now: 

361 words_we_already_visited |= link_group 

362 

363 # second iteration is to actually do the linking: 

364 words_we_already_visited: Set[str] = set() 

365 for word_name in graph.keys(): 

366 # if we did not visit this word yet: 

367 if word_name not in words_we_already_visited: 

368 link_group = link_groups[word_name] 

369 words_we_already_visited |= link_group 

370 

371 # create a gender-dict for all of them: 

372 genders_we_are_very_sure_about: Set[GeneratedDataGender] = set() 

373 gender_dict: Dict[GeneratedDataGender, str] = dict() 

374 gender_dict_alts: Dict[GeneratedDataGender, Set[str]] = {"female": set(), "male": set(), 

375 "neutral": set()} 

376 for link_name in link_group: 

377 # add gender of the individual... 

378 gender = graph[link_name]["gender"] 

379 if gender in genders_we_are_very_sure_about: 

380 assign_value_or_use_old_one(gender_dict, gender, link_name) 

381 else: 

382 gender_dict_alts[gender] = {link_name} 

383 gender_dict[gender] = link_name 

384 genders_we_are_very_sure_about.add(gender) 

385 

386 # ...as well as genders it specifies in its gender_map, but only if we couldn't find these otherwise 

387 for gender, link_name2 in graph[link_name]["gender_map"].items(): 

388 if gender not in genders_we_are_very_sure_about: 

389 assign_value_or_use_old_one(gender_dict, gender, link_name2) 

390 

391 # link words with each other, so each word gets the full gender dict to call its own, 

392 # but only for genders it does not have in its own original gender dict yet, so manual links in the 

393 # already given data take precedence: 

394 for link_name in link_group: 

395 for gender, link_name2 in gender_dict.items(): 

396 if gender not in graph[link_name]["gender_map"] and gender != graph[link_name]["gender"]: 

397 graph[link_name]["gender_map"][gender] = link_name2 

398 lwarn("\"" + link_name + "\" is (indirectly) linked to \"" + link_name2 

399 + "\", which is " + graph[link_name2]["gender"], " but \"" + link_name 

400 + "\" has no " + graph[link_name2]["gender"] + " version.") 

401 count += 1 

402 

403 # if the newly added link was chosen between two possible words algorithmically, add this as 

404 # a warning: 

405 if gender_dict_alts[gender] != {link_name2}: 

406 if "warning" not in graph[link_name]: 

407 graph[link_name]["warning"] = set() 

408 alt_values = sorted(list(gender_dict_alts[gender])) 

409 alt_values_str = ", ".join(alt_values[:-1]) + " and " + alt_values[-1] 

410 new_warning = (alt_values_str + " would've all been good values for the " + gender + " " 

411 + "version of \"" + link_name + "\", but \"" + link_name2 

412 + "\" was automatically chosen based on an algorithm.") 

413 lwarn(new_warning) 

414 graph[link_name]["warning"].add(new_warning) 

415 

416 # infect linked words with the warnings of the words they're linked to: 

417 link_group_warnings = set() 

418 for link_name in link_group: 

419 if "warning" in graph[link_name]: 

420 link_group_warnings |= graph[link_name]["warning"] 

421 if link_group_warnings: 

422 for link_name in link_group: 

423 if "warning" not in graph[link_name]: 

424 graph[link_name]["warning"] = set() 

425 graph[link_name]["warning"] |= link_group_warnings 

426 

427 lwarn(count, "links created.") 

428 return graph

502 not_applicable = False 

503 if gender != "neutral": 

504 other_gender_indicator_tuples = [

510 new_gendered_version = (word_name[:-len(gender_indicator)] 

511 + other_gender_indicator) 

512 else: 

513 new_gendered_version = (other_gender_indicator 

514 + word_name[len(gender_indicator):])

546

548 created_corresponding_gendered_versions = True

684 def render_noun(self, gender: GeneratedDataGender) -> str: 

685 """Returns the correctly gendered version of itself as a string. g must be either "male", "female" or 

686 "neutral". 

687 Capitalisation of the first letter is identical between out- and input.""" 

688 # remember if the first letter of the word is uppercase: 

689 if self.word[0].isupper(): # ToDo: Maybe add support for all-caps-writing? What about words with "_"in them? 

690 uppercase = True 

691 else: 

692 uppercase = False 

693 

694 # return the correctly gendered version of the word: 

695 if self.word.lower() in GENDER_DICT: 

696 word = self.word.lower() 

697 word_data = GENDER_DICT[word] 

698 # look for the neutral version if there is no version of the given gender: 

699 if gender not in set(word_data["gender_map"].keys()) | {word_data["gender"]}: 

700 gender = "neutral" 

701 # return the word if it is the right gender: 

702 if GENDER_DICT[word]["gender"] == gender: 

703 result = word 

704 # otherwise, return the correctly gendered version from the gender_map: 

705 else: 

706 result = word_data["gender_map"][gender] 

707 # make the first letter of the result uppercase in case we had to lowercase it before: 

708 if uppercase: 

709 result = result[0].upper() + result[1:] 

710 else: 

711 result = self.word 

712 return result.replace("_", " ")
